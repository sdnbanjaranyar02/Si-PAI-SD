    <script>
        // -- CORE ENGINE --
        const state = {
            isLoggedIn: localStorage.getItem('sipai_auth') === 'true',
            profil: JSON.parse(localStorage.getItem('sipai_profil')) || {
                nama: 'Guru PAI', nip: '-', unit: 'SD NEGERI -', jabatan: 'Guru PAI', foto: ''
            },
            siswa: JSON.parse(localStorage.getItem('sipai_siswa')) || [],
            jurnal: JSON.parse(localStorage.getItem('sipai_jurnal')) || []
        };

        function sync() {
            localStorage.setItem('sipai_auth', state.isLoggedIn);
            localStorage.setItem('sipai_profil', JSON.stringify(state.profil));
            localStorage.setItem('sipai_siswa', JSON.stringify(state.siswa));
            localStorage.setItem('sipai_jurnal', JSON.stringify(state.jurnal));
            render();
        }

        // -- AUTH --
        function authAction() {
            const u = document.getElementById('user-input').value;
            const p = document.getElementById('pass-input').value;
            if(u === 'admin' && p === 'admin') {
                state.isLoggedIn = true;
                sync();
            } else { alert('Gagal! Gunakan admin/admin'); }
        }

        function logout() {
            if(confirm('Keluar aplikasi?')) {
                state.isLoggedIn = false;
                sync();
            }
        }

        // -- NAV --
        function nav(tabId) {
            document.querySelectorAll('.tab-content').forEach(t => {
                t.classList.replace('active-tab', 'hidden-tab');
            });
            document.getElementById('tab-' + tabId).classList.replace('hidden-tab', 'active-tab');
            
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.replace('text-emerald-600', 'text-slate-400');
            });
            event.currentTarget.classList.replace('text-slate-400', 'text-emerald-600');
            window.scrollTo(0,0);
        }

        // -- PROFIL --
        function handleImage(input) {
            if (input.files && input.files[0]) {
                const reader = new FileReader();
                reader.onload = e => {
                    state.profil.foto = e.target.result;
                    document.getElementById('preview-foto').src = e.target.result;
                };
                reader.readAsDataURL(input.files[0]);
            }
        }

        function saveProfil() {
            state.profil.nama = document.getElementById('set-nama').value;
            state.profil.nip = document.getElementById('set-nip').value;
            state.profil.unit = document.getElementById('set-unit').value;
            state.profil.jabatan = document.getElementById('set-jabatan').value;
            sync();
            alert('Profil disimpan!');
            nav('home');
        }

        // -- JURNAL --
        function openJurnalModal() { document.getElementById('modal-jurnal').style.display = 'flex'; }
        function openSiswaModal() { document.getElementById('modal-siswa').style.display = 'flex'; }
        function closeModal(id) { document.getElementById(id).style.display = 'none'; }

        function saveJurnal() {
            const data = {
                id: Date.now(),
                tgl: document.getElementById('jur-tgl').value,
                kls: document.getElementById('jur-kls').value,
                materi: document.getElementById('jur-materi').value,
                ket: document.getElementById('jur-ket').value
            };
            if(!data.materi) return;
            state.jurnal.unshift(data);
            closeModal('modal-jurnal');
            sync();
        }

        function delJurnal(id) {
            state.jurnal = state.jurnal.filter(j => j.id !== id);
            sync();
        }

        // -- SISWA --
        function saveSiswa() {
            const nama = document.getElementById('sis-nama').value;
            const kls = document.getElementById('sis-kls').value;
            if(!nama) return;
            state.siswa.push({ id: Date.now(), nama, kls, nilai: 0 });
            closeModal('modal-siswa');
            sync();
        }

        function delSiswa(id) {
            state.siswa = state.siswa.filter(s => s.id !== id);
            sync();
        }

        // -- RENDERER --
        function render() {
            const login = document.getElementById('login-page');
            const app = document.getElementById('app-container');

            if(!state.isLoggedIn) {
                login.classList.remove('hidden');
                app.classList.add('hidden');
                return;
            }
            login.classList.add('hidden');
            app.classList.remove('hidden');

            // Dashboard
            document.getElementById('view-nama').innerText = state.profil.nama;
            document.getElementById('view-nip').innerText = 'NIP. ' + state.profil.nip;
            document.getElementById('view-unit').innerText = state.profil.unit;
            if(state.profil.foto) {
                document.getElementById('view-foto').src = state.profil.foto;
                document.getElementById('preview-foto').src = state.profil.foto;
            }
            document.getElementById('count-siswa').innerText = state.siswa.length;
            document.getElementById('count-jurnal').innerText = state.jurnal.length;
            document.getElementById('date-now').innerText = new Date().toLocaleDateString('id-ID', {weekday:'long', day:'numeric', month:'long', year:'numeric'});

            // Profil Inputs
            document.getElementById('set-nama').value = state.profil.nama;
            document.getElementById('set-nip').value = state.profil.nip;
            document.getElementById('set-unit').value = state.profil.unit;
            document.getElementById('set-jabatan').value = state.profil.jabatan;

            // Jurnal List
            document.getElementById('jurnal-list').innerHTML = state.jurnal.map(j => `
                <div class="bg-white p-6 rounded-[2rem] border border-slate-100 relative group">
                    <button onclick="delJurnal(${j.id})" class="absolute top-4 right-4 text-slate-200 hover:text-red-400 transition-colors"><i class="fas fa-trash-alt"></i></button>
                    <span class="text-[9px] font-black text-emerald-600 bg-emerald-50 px-2 py-1 rounded-lg uppercase">Kelas ${j.kls}</span>
                    <p class="text-[10px] text-slate-400 font-bold mt-2">${j.tgl}</p>
                    <h3 class="font-bold text-slate-800 mt-2 mb-2 leading-tight">${j.materi}</h3>
                    <p class="text-xs text-slate-500 leading-relaxed italic line-clamp-3">${j.ket}</p>
                </div>
            `).join(state.jurnal.length === 0 ? '<p class="col-span-2 text-center p-12 text-slate-300 italic">Belum ada kegiatan</p>' : '');

            // Siswa List
            document.getElementById('siswa-list').innerHTML = state.siswa.map(s => `
                <tr class="hover:bg-slate-50">
                    <td class="p-4 font-bold text-sm">${s.nama}</td>
                    <td class="p-4 text-center"><span class="bg-blue-50 text-blue-600 text-[10px] font-black px-2 py-1 rounded-md">KLS ${s.kls}</span></td>
                    <td class="p-4 text-center font-black text-emerald-600">${s.nilai}</td>
                    <td class="p-4 text-center">
                        <button onclick="delSiswa(${s.id})" class="text-slate-200 hover:text-red-500"><i class="fas fa-trash-alt"></i></button>
                    </td>
                </tr>
            `).join(state.siswa.length === 0 ? '<tr><td colspan="4" class="p-10 text-center text-slate-300 italic">Belum ada siswa</td></tr>' : '');
        }

        // Service Worker (Offline Support)
        if ('serviceWorker' in navigator) {
            window.addEventListener('load', () => {
                const blob = new Blob([`
                    self.addEventListener('install', e => self.skipWaiting());
                    self.addEventListener('fetch', e => e.respondWith(fetch(e.request).catch(() => caches.match(e.request))));
                `], {type: 'text/javascript'});
                navigator.serviceWorker.register(URL.createObjectURL(blob));
            });
        }

        window.onload = render;
    </script>
